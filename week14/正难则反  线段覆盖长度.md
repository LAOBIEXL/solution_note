#include<iostream>

using namespace std;
typedef long long ll;
typedef unsigned long long ull;
typedef pair<int, int> PII;
const int N = 2e5 + 10;
int sum_pre[N], a[N];

bool cmp(int a, int b)
{
    return a > b;
}

signed main()
{
    int n;
    cin >> n;
    int sum_all = 0;
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
        if (i == 1) continue;
        sum_pre[i - 1] = a[i] - a[i - 1];
        sum_all += sum_pre[i - 1];
    }
    sort(sum_pre + 1, sum_pre + n, cmp);
    for (int j = 1; j <= n; j++)
    {
        cout << (sum_all = sum_all - sum_pre[j - 1]) << " ";
    }
    cout << endl;
}
