这个题目不是正常的子串，即即使三个字母不是连续出现，只要顺序是对的就计数；
方法是对每一次加入都动态更新相应的变量；
![[Pasted image 20251224205443.png]]
所以这个题最关键的地方起始就是理解相对位置，而不是广义的子串；

==相对顺序==
```cpp
#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<string>
#include<cstring>
#include<vector>
#include<set>
#include<map>
#include<unordered_map>
#include<unordered_set>
#include<algorithm>
#include<cmath>
#include<queue>
#include <deque>
#include <stack>
#include<iomanip>
#include <chrono>
#include<random>
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
typedef pair<int, int> PII;
#define endl '\n'

ll mod = 1e9 + 7;

void solve()
{
    //不要忘记清空数组；
    string s;
    cin >> s;
    ull a = 0, c = 0, m = 0, ac = 0, cm = 0, acm = 0;
    for (int i = 0; i < s.size(); i++)
    {
        if (s[i] == 'A')
        {
            a++;
            a %= mod;
        }
        if (s[i] == 'a')
        {
            acm += cm;
            acm %= mod;
            ac += c;
            ac %= mod;
            a++;
            a %= mod;
        }
        if (s[i] == 'C')
        {
            ac += a;
            ac %= mod;
            c++;
            c %= mod;
        }
        if (s[i] == 'c')
        {
            cm += m;
            cm %= mod;
            c++;
            c %= mod;
        }
        if (s[i] == 'M')
        {
            acm += ac;
            acm %= mod;
            m++;
            m %= mod;
            cm += c;
            cm %= mod;
        }
        if (s[i] == 'm')
        {
            m++;
            m %= mod;

        }
        cout << acm % mod << " ";
    }
    cout << endl;
}

signed main()
{
    ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
    int TestCase = 1;
    cin >> TestCase;
    while (TestCase--)
        solve();
}
```
